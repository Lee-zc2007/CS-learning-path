class struct union虽然都属于类，但是只有为class和struct时is_class为真，union需要单独用is_union

std::false_type
std::trus_type
可以用于简化类模板
'''
//原来
template<typename T>
struct my_is_integral{
    static const bool value{false};
};

template<>
struct my_is_integral<int>{
    static const bool value{true};
};

//用了true_type
template<>
struct my_is_integral<int>: std::true_type{};
'''

*简化调用*
c++14后增加了变量模板，可以通过定义变量模板简化调用代码
'''
template <typename T>
constexpr bool my_is_integral_v{my_is_integral<T>::value};

std::cout<<my_is_integral_v<long><<std::endl;
'''

但是这种优化对有cv修饰符无法得到正确结果，因此需要std::remove_cv
'''
template <typename T>
struct my_is_integral_helper: std::flase_type{};

template <>
struct my_is_integral_helper<int>: std::true_type{};

template <typename T>
struct my_is_integral: my_is_integral_helper<type std::remove_cv<T>::type>{};

template <typename T>
constexpr bool my_is_integral_v{my_is_integral<T>::value};
'''

type_traits中的模板变量都是常量，可以用于编译时的条件判断（if constexpr，static_assert）
